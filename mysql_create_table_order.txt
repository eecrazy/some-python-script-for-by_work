#建立总表：词-时间-报纸-词集（词\t个数\t）
'''

CREATE DATABASE by_work DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;

drop table paper_all;

CREATE TABLE paper_simple
(
word varchar(210) NOT NULL,
month int NOT NULL,
count int NOT NULL,
word_set MEDIUMTEXT NOT NULL
);


CREATE TABLE paper_single_word
(
word varchar(210) NOT NULL,
month int NOT NULL,
count int NOT NULL,
word_set MEDIUMTEXT NOT NULL
);

'''
